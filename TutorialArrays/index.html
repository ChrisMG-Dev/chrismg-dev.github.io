Ejercicio 3

  • Create a function filterNumeric(arr) which takes an array and returns new array
    which contains only numeric values from arr.
                var arr = ["Uno", 1, "Dos", 2];

                function filterNumeric(arr) {
                    var newArr = [],
                        i;
                    for (i = 0; i < arr.length; i += 1) {
                        if (!isNaN(arr[i])) {
                            newArr.push(arr[i]);
                        }
                    }
                    return newArr;
                }

                filterNumeric(arr) //[1, 2]

Ejercicio 4

  • An object has a className property which keeps it’s class names delimited by spaces:
                    var obj = {
                    className: 'open menu'
                }
  • Write a function addClass(obj, cls) which adds a class cls, but only if it doesn’t yet exist
            var obj = {
                className: 'open menu'
            }

            function addClass(obj, cls) {
                var spObj,
                    i;

                spObj = obj.className.split(" ");

                for (i in spObj) {
                    if (spObj[i] === cls) {
                        return;
                    }
                }
                obj.className += " " + cls;
            }

            addClass(obj, "menu"); // className: 'open menu';
            addClass(obj, "hola"); // className: 'open menu hola';

Ejercicio 5

  • Create a function camelize(str) which transforms a string from
    “my-short-string” to “myShortString”. 
                camelize("background-color") == 'backgroundColor'
            camelize("list-style-image") == 'listStyleImage'
                
            var dashedString = "my-short-string";

            function cameliza(str) {
                var camelized = dashedString.split("-"),
                    normalizedString = [camelized[0]],
                    i;
                for (i = 1; i < camelized.length; i += 1) {
                    normalizedString.push(camelized[i].charAt(0).toUpperCase() + camelized[i].substring(1));
                }

                return normalizedString.join("");
            }
